<p>Converted date: <span id="outdate"></span></p>

$('#dodate').live('vclick', function() {
    $('#dbox').data('datebox').options.dateFormat = 'YYYY-MM-DD'; // INPUT Format
    $('#dbox').val('2001-03-04').trigger('change'); // Value to convert
    
    // Call _formatter with the OUTPUT format...
    $('#outdate').text($('#dbox').data('datebox')._formatter('ddd, mmm ddSS YYYY', $('#dbox').data('datebox').theDate));
})
